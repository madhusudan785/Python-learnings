class Complex:
    def __init__(self,real,immg):
        self.real=real
        self.immg=immg
    def showNumber(self):
        print(self.real,"i +",self.immg,"j")
    def __add__(self,num2):
        newReal=self.real+num2.real
        newImmg=self.immg+num2.immg
        return Complex(newReal,newImmg)

num=Complex(2,3)
num.showNumber()

num2.showNumber()
